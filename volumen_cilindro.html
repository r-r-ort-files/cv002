#Programa: Volumen de un cilindro
r=float(input("ingresa el valor del radio de la base del cilindro: "))
pi=3.1416
B=pi*(r**2)

print("el area de la base del cilindro es: ", B)
L=float(input("ingresa el valor del largo del cilindro"))

V= L*B
print("el volumen del cilindro es: ", V)
